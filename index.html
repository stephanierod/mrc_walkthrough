ld <- language_diversity %>% 
  mutate_if(is.character, as.factor) %>% 
  pivot_wider(
    names_from = Measurement, 
    values_from = Value
  ) %>% 
  mutate(
    lang_log = log(Langs), 
    area_log = log(Area), 
    pop_log = log(Population)
  )
